Project README Modules @fliegwerk/trc

@fliegwerk/trc

Install using 

npm install @fliegwerk/trc

Exported Members

You can access these members by importing the module:

import { member } from '@fliegwerk/trc';

build

function build (
   webpackConfig : Configuration
) : Promise<void>

Builds project source files with webpack via a given configuration file.

@param:

webpackConfig - the configuration file for the webpack compiler

@example: 

import { getWebpackConfig, getPaths, build } from '@fliegwerk/trc';

const paths = getPaths();
const config = getWebpackConfig(paths);
build(config).then(() => console.log('Successfully built'));

watch

function watch (
   webpackConfig : Configuration ,
   devServerConfig : DevServerConfiguration
) : void

Builds and serves a webpack compiled project. It uses the webpack development server to compile and serve the built project sources. Both the webpack compiler and development server can be configured via configuration files.

@param:

webpackConfig - the configuration file for the webpack compiler used by the development server

@param:

devServerConfig - the configuration file the webpack development server

@example: 

import { getDevServerConfig, watch } from '@fliegwerk/trc';
import Server from 'webpack-dev-server';

const paths = getPaths();
const config = getWebpackConfig(paths);
const devServerConfig = getDevServerConfig();

watch(config, devServerConfig);

getLogger

function getLogger (
   component : string
) : ComponentLogger

Returns a logsemts ComponentLogger for the given component.

This provides the capability of logging in a style consistent with the one of trc

@param:

component - the component for which the logger gets created

@returns:

the ComponentLogger for component

@example: 

const logger = getLogger('my-componenet');

logger.success('A success message');
logger.warn('A warning message');
logger.error('An error message');
logger.info('An information');

// gets printed when DEBUG is set to a pattern matching 'my-component'
logger.debug('Some debug information');

getPaths

function getPaths ( ) : Paths

Returns a set of paths commonly required for creating a webpack configuration.

Gets passed to getWebpackConfig.

@returns:

the paths required for creating a webpack config

@example: 

const paths = getPaths();
const webpackConfig = createWebpackConfig(paths, isDev);

createCompiler

function createCompiler (
   webpackConfig : Configuration
) : Compiler

Creates a webpack.Compiler, based on the webpackConfig

@param:

webpackConfig - the webpack configuration

@example: 

const config = getWebpackConfig(paths, isDev);
const compiler = createCompiler(config);
compiler.run(() => {
    compiler.close();
});

getDevServerConfig

function getDevServerConfig ( ) : Configuration

Returns the webpack development server configuration which can be used in a webpack development server.

@example: 

import { getDevServerConfig, watch } from '@fliegwerk/trc';
import Server from 'webpack-dev-server';

const paths = getPaths();
const config = getWebpackConfig(paths);
const devServerConfig = getDevServerConfig();

watch(config, devServerConfig);

getWebpackConfig

function getWebpackConfig (
   paths : Paths ,
   isDev : boolean = true
) : Configuration

Builds and returns a webpack configuration for a webpack compiler which compiles TypeScript React Code into browser runnable JavaScript.

@param:

paths - the paths needed to build this configuration (e.g. project root, output directory, ...)

@param:

isDev - when true webpack will build in development mode and when false in production mode

@example: 

import { getWebpackConfig, getPaths, build } from '@fliegwerk/trc';

const paths = getPaths();
const config = getWebpackConfig(paths);
build(config).then(() => console.log('Successfully built'));
